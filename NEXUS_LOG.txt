class GrogKernel {
  private _commands: Map<string, (args: any) => void>;
  private _queries: Map<string, (args: any) => void>;
  private _errors: Map<string, (args: any) => void>;

  constructor() {
    this._commands = new Map();
    this._queries = new Map();
    this._errors = new Map();
  }

  registerCommand(name: string, callback: (args: any) => void) {
    this._commands.set(name, callback);
  }

  registerQuery(name: string, callback: (args: any) => void) {
    this._queries.set(name, callback);
  }

  registerError(name: string, callback: (args: any) => void) {
    this._errors.set(name, callback);
  }

  getCommand(name: string) {
    return this._commands.get(name);
  }

  getQuery(name: string) {
    return this._queries.get(name);
  }

  getError(name: string) {
    return this._errors.get(name);
  }
}
