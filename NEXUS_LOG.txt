interface ErrorHandler {
  handle(error: Error): void;
}

class ErrorHandlerImpl implements ErrorHandler {
  private logger: Logger;

  constructor(logger: Logger) {
    this.logger = logger;
  }

  public handle(error: Error): void {
    this.logger.error(error.message);
  }
}

interface EvolutionStep {
  step: string;
  callback: () => Promise<void>;
}

class EvolutionStepImpl {
  private steps: EvolutionStep[];

  constructor(steps: EvolutionStep[]) {
    this.steps = steps;
  }

  public async execute(): Promise<void> {
    for (const step of this.steps) {
      await step.callback();
    }
  }
}

class StrategicEvolutionImpl {
  private evolutionSteps: EvolutionStep[];

  constructor(evolutionSteps: EvolutionStep[]) {
    this.evolutionSteps = evolutionSteps;
  }

  public async evolve(): Promise<void> {
    await this.evolutionSteps.execute();
  }
}

interface PubSub {
  subscribe(subscriber: (event: any) => void): void;
  unsubscribe(subscriber: (event: any) => void): void;
}

class PubSubImpl implements PubSub {
  private subscribers: (event: any) => void[];

  constructor() {
    this.subscribers = [];
  }

  public subscribe(subscriber: (event: any) => void): void {
    this.subscribers.push(subscriber);
  }

  public unsubscribe(subscriber: (event: any) => void): void {
    const index = this.subscribers.indexOf(subscriber);
    if (index !== -1) {
      this.subscribers.splice(index, 1);
    }
  }

  public publish(event: any): void {
    for (const subscriber of this.subscribers) {
      subscriber(event);
    }
  }
}
